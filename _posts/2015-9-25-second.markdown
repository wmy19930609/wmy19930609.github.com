关于手机的判断

```javascript
function isIphone(){
	var u = navigator.userAgent;
	if(u.indexOf('iPhone') > -1){
		return true;
	}else{
		return false;
	} 
}
function isAndroid(){
	var u = navigator.userAgent;
	if(u.indexOf('Android') > -1){
		return true;
	}else{
		return false;
	} 
}

```

是否是微信

```javascript
function isWX(){
    var ua = navigator.userAgent;
    if(ua.indexOf("MicroMessenger") > -1) {
        return true;
    } else {
        return false;
    }
}
```

是否是safari

```javasscript
function isSafari(){
	var u = navigator.userAgent.toLowerCase();
	var isIphoneFlag = isIphone();
	if(isIphoneFlag && u.indexOf('safari') > -1){
		return true;
	}else{
		return false;
	}
}
```
