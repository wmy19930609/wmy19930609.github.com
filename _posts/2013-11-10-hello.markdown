**第一题**

```javascript
function reverseWords(str){
	var aStr = str.split(/\s+/);
	aStr.reverse();
	newStr = aStr.join(' ');
	return newStr;
}
console.log(reverseWords('hello World'));
```

```javascript
<!DOCTYPE html>
<html lang="en">
<head>
	<meta charset="UTF-8">
	<title>random</title>
	<style>
		#num{
			width: 35px;
			height: 35px;
			font-size: 34px;
			text-align: center;
			color: red;
		}
	</style>
</head>
<body>
	<input type="button" value="开始" />
	<div id="num"></div>
	<script>
		function $(selector){
			return document.querySelector(selector);
		}
		var startbtn = $('input');
		var num = $('#num');

		startbtn.onclick = function(){
			startRandom([5,6], 5000, 500);
		};

		function startRandom(arr, time, interval){
			var len = time/interval;
			var count = 0;
			
			var timer = setInterval(function(){
				count ++;
				var ranIndex = parseInt(Math.random() * arr.length);
				num.innerText = arr[ranIndex];
				if(count >= len){
					clearInterval(timer);
					alert('it is over');
				}
			}, interval);
		}
	</script>
</body>
</html>
```
